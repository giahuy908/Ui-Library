-- CÁC HÀM TRỢ GIÚP TIỆN ÝCH (HELPERS)
local function CreateTween(obj, info, properties)
    local tween = TweenService:Create(obj, info, properties)
    tween:Play()
    return tween
end

local function MakeDraggable(dragHandle, mainObject)
    local dragging = false
    local dragInput, dragStart, startPos

    dragHandle.InputBegan:Connect(function(input)
        -- Ngăn chặn kéo UI khi đang ở chế độ phóng to toàn màn hình
        if mainObject:GetAttribute("Maximized") then return end

        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = mainObject.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    dragHandle.InputChanged:Connect(function(input)
        if mainObject:GetAttribute("Maximized") then return end

        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if mainObject:GetAttribute("Maximized") then return end

        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            mainObject.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

-- TẠO GRADIENT ĐỘNG
function NeoLib:CreateGradient(parent, colors, rotation)
    local gradient = Instance.new("UIGradient")
    gradient.Rotation = rotation or 45
    local keypoints = {}
    for pos, color in pairs(colors) do
        table.insert(keypoints, ColorSequenceKeypoint.new(pos/100, color))
    end
    table.sort(keypoints, function(a, b) return a.Time < b.Time end)
    gradient.Color = ColorSequence.new(keypoints)
    gradient.Parent = parent
    return gradient
end

            if ActiveTab then
                -- Bỏ chọn tab cũ
                CreateTween(ActiveTab, TweenInfo.new(0.3), {BackgroundTransparency = 1})
                CreateTween(ActiveTab.AccentBar, TweenInfo.new(0.3), {BackgroundTransparency = 1})
                CreateTween(ActiveTab.TabTitle, TweenInfo.new(0.3), {TextColor3 = theme.SubText})
                CreateTween(ActiveTab.TabIcon, TweenInfo.new(0.3), {ImageColor3 = theme.SubText})

            CreateTween(TabBtn, TweenInfo.new(0.3), {BackgroundTransparency = 0.8, BackgroundColor3 = theme.Accent})
            CreateTween(AccentBar, TweenInfo.new(0.3), {BackgroundTransparency = 0})

                rounding = rounding or 0

                ValLabel.TextColor3 = theme.Accent
                ValLabel.TextSize = 13
                ValLabel.TextXAlignment = Enum.TextXAlignment.Right
                ValLabel.Parent = Slider

                -- Khung trượt chính
                local SlideBar = Instance.new("Frame")
                SlideBar.Size = UDim2.new(1, 0, 0, 6)
                SlideBar.Position = UDim2.new(0, 0, 0, 32)
                SlideBar.BackgroundColor3 = theme.Border
                SlideBar.Parent = Slider

                local SlideBarCorner = Instance.new("UICorner")
                SlideBarCorner.CornerRadius = UDim.new(1, 0)
                SlideBarCorner.Parent = SlideBar

                -- Thanh đo đã kéo màu sắc nổi bật
                local Fill = Instance.new("Frame")
                Fill.Size = UDim2.new((value - min) / (max - min), 0, 1, 0)
                Fill.BackgroundColor3 = theme.Accent
                Fill.BorderSizePixel = 0
                Fill.Parent = SlideBar

                local FillCorner = Instance.new("UICorner")
                FillCorner.CornerRadius = UDim.new(1, 0)
                FillCorner.Parent = Fill

                -- Nút kéo
                local Knob = Instance.new("Frame")
                Knob.Size = UDim2.new(0, 14, 0, 14)
                Knob.AnchorPoint = Vector2.new(0.5, 0.5)
                Knob.Position = UDim2.new((value - min) / (max - min), 0, 0.5, 0)
                Knob.BackgroundColor3 = theme.TextColor
                Knob.Parent = SlideBar

                local KnobCorner = Instance.new("UICorner")
                KnobCorner.CornerRadius = UDim.new(1, 0)
                KnobCorner.Parent = Knob

                local KnobStroke = Instance.new("UIStroke")
                KnobStroke.Color = theme.Accent
                KnobStroke.Thickness = 1.5
                KnobStroke.Parent = Knob

                -- HỆ THỐNG KÉO THẢ TUYỆT VỜI
                local dragging = false

                local function UpdateValueFromInput(input)
                    local percent = math.clamp((input.Position.X - SlideBar.AbsolutePosition.X) / SlideBar.AbsoluteSize.X, 0, 1)
                    local rawVal = min + (max - min) * percent
                    value = tonumber(string.format("%."..rounding.."f", rawVal))
                    
                    ValLabel.Text = tostring(value)
                    CreateTween(Fill, TweenInfo.new(0.1), {Size = UDim2.new(percent, 0, 1, 0)})
                    CreateTween(Knob, TweenInfo.new(0.1), {Position = UDim2.new(percent, 0, 0.5, 0)})
                    
                    task.spawn(callback, value)
                end

                SlideBar.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                        dragging = true
                        UpdateValueFromInput(input)
                    end
                end)

                UserInputService.InputChanged:Connect(function(input)
                    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                        UpdateValueFromInput(input)
                    end
                end)

                UserInputService.InputEnded:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                        dragging = false
                    end
                end)

                local ctrl = {Name = sliderText, Element = Slider, Section = SectionFrame, SectionName = sectionName, TabName = tabName}
                table.insert(getgenv().AllControls, ctrl)

                return {
                    SetValue = function(val)
                        value = math.clamp(val, min, max)
                        local percent = (value - min) / (max - min)
                        ValLabel.Text = tostring(value)
                        CreateTween(Fill, TweenInfo.new(0.2), {Size = UDim2.new(percent, 0, 1, 0)})
                        CreateTween(Knob, TweenInfo.new(0.2), {Position = UDim2.new(percent, 0, 0.5, 0)})
                        task.spawn(callback, value)
                    end
                }
            end

                    end
                }

            CreateTween(SearchBoxStroke, TweenInfo.new(0.4), {Color = theme.Border})
            CreateTween(SearchInput, TweenInfo.new(0.4), {TextColor3 = theme.TextColor})
            CreateTween(UserProfile, TweenInfo.new(0.4), {BackgroundColor3 = theme.Card})
            CreateTween(Username, TweenInfo.new(0.4), {TextColor3 = theme.TextColor})
            CreateTween(UserRole, TweenInfo.new(0.4), {TextColor3 = theme.Accent})

            -- Đổi màu các biểu tượng hệ thống topbar đồng bộ với theme mới
            Logo.ImageColor3 = theme.Accent
            MinimizeBtn.ImageColor3 = theme.TextColor
            MaximizeBtn.ImageColor3 = theme.TextColor
            CloseBtn.ImageColor3 = theme.TextColor
        end
    end

    -- Nút Đổi Theme nhanh chóng
    local ThemeToggleBtn = Instance.new("ImageButton")
    ThemeToggleBtn.Size = UDim2.new(0, 24, 0, 24)
    ThemeToggleBtn.BackgroundTransparency = 1
    ThemeToggleBtn.Image = "rbxassetid://3926305904"
    ThemeToggleBtn.ImageRectOffset = Vector2.new(924, 204)
    ThemeToggleBtn.ImageRectSize = Vector2.new(24, 24)
    ThemeToggleBtn.ImageColor3 = theme.TextColor
    ThemeToggleBtn.Parent = TopbarButtons

    local themeNames = {"Dark", "Light", "Cyberpunk"}
    local themeIndex = 1
    ThemeToggleBtn.MouseButton1Click:Connect(function()
        themeIndex = themeIndex + 1
        if themeIndex > #themeNames then themeIndex = 1 end
        SwitchTheme(themeNames[themeIndex])
        
        NeoLib:Notify({
            Title = "Đổi Chủ Đề",
            Content = "Đã áp dụng giao diện: " .. themeNames[themeIndex],
            Icon = "rbxassetid://3926305904"
        })
    end)

    -- NÚT NỔI BẬT KHỞI ĐỘNG TRÒN TRÊN MÀN HÌNH (FLOATING/OPEN TOGGLE BUTTON)
    local OpenButtonGui = Instance.new("ScreenGui")
    OpenButtonGui.Name = "NeoLib_Toggle"
    OpenButtonGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    OpenButtonGui.Parent = CoreGui

    local FloatingBtn = Instance.new("ImageButton")
    FloatingBtn.Size = UDim2.new(0, 50, 0, 50)
    FloatingBtn.Position = UDim2.new(0, 20, 0.5, -25)
    FloatingBtn.BackgroundColor3 = theme.Sidebar
    FloatingBtn.AutoButtonColor = false
    FloatingBtn.Parent = OpenButtonGui

    local FloatingCorner = Instance.new("UICorner")
    FloatingCorner.CornerRadius = UDim.new(1, 0)
    FloatingCorner.Parent = FloatingBtn

    local FloatingStroke = Instance.new("UIStroke")
    FloatingStroke.Color = theme.Accent
    FloatingStroke.Thickness = 2
    FloatingStroke.Parent = FloatingBtn

    local FloatingIcon = Instance.new("ImageLabel")
    FloatingIcon.Size = UDim2.new(0, 30, 0, 30)
    FloatingIcon.Position = UDim2.new(0.5, -15, 0.5, -15)
    FloatingIcon.BackgroundTransparency = 1
    FloatingIcon.Image = config.Logo or "rbxassetid://88031262069243"
    FloatingIcon.ImageColor3 = theme.Accent
    FloatingIcon.Parent = FloatingBtn

    MakeDraggable(FloatingBtn, FloatingBtn)

    -- ĐÓNG / MỞ HOẠT ẢNH MƯỢT MÀ CHO WINDOW CHÍNH
    local isOpened = true
    
    local function ToggleUI()
        isOpened = not isOpened
        if isOpened then
            MainFrame.Visible = true
            CreateTween(MainFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                Size = MainFrame:GetAttribute("Maximized") and UDim2.new(1, 0, 1, 0) or size,
                BackgroundTransparency = 0
            })
            CreateTween(FloatingBtn, TweenInfo.new(0.3), {BackgroundColor3 = theme.Sidebar})
        else
            local closeTween = CreateTween(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.In), {
                Size = UDim2.fromOffset(0, 0),
                BackgroundTransparency = 1
            })
            closeTween.Completed:Connect(function()
                if not isOpened then MainFrame.Visible = false end
            end)
            CreateTween(FloatingBtn, TweenInfo.new(0.3), {BackgroundColor3 = theme.Card})
        end
    end

    FloatingBtn.MouseButton1Click:Connect(ToggleUI)
    MinimizeBtn.MouseButton1Click:Connect(ToggleUI)

    CloseBtn.MouseButton1Click:Connect(function()
        NeoLib:ShowDialog({
            Title = "Thoát Menu?",
            Content = "Bạn có chắc chắn muốn đóng hoàn toàn và hủy giao diện NeoLib không?",
            Buttons = {
                {
                    Title = "Hủy",
                    Variant = "Secondary"
                },
                {
                    Title = "Đồng ý",
                    Variant = "Primary",
                    Callback = function()
                        ScreenGui:Destroy()
                        OpenButtonGui:Destroy()
                    end
                }
            }
        })
    end)

    -- SỰ KIỆN TÌM KIẾM ĐỘNG TRÊN SIDEBAR (BANANAHUB GLOBAL SEARCH)
    SearchInput:GetPropertyChangedSignal("Text"):Connect(function()
        local query = string.lower(SearchInput.Text)
        for _, control in ipairs(getgenv().AllControls) do
            local ctrlName = string.lower(control.Name)
            local sectionNameLower = string.lower(control.SectionName)
            
            if query == "" then
                control.Element.Visible = true
                control.Section.Visible = true
            else
                if string.find(ctrlName, query) or string.find(sectionNameLower, query) then
                    control.Element.Visible = true
                    control.Section.Visible = true
                else
                    control.Element.Visible = false
                end
            end
        end
    end)

    -- Hiệu ứng xuất hiện lúc ban đầu cực đẹp
    MainFrame.Size = UDim2.fromOffset(0, 0)
    CreateTween(MainFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Size = size})

    return {
        AddTab = function(_, tabName, iconId) return NeoLib:AddTab(tabName, iconId) end,
        AddTag = function(_, tagName, color) return NeoLib:AddTag(tagName, color) end,
        ShowDialog = function(_, config) return NeoLib:ShowDialog(config) end,
        Notify = function(_, config) return NeoLib:Notify(config) end
    }
end

return NeoLib
