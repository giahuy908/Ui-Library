local function CreateTween(obj, info, properties)
    local tween = TweenService:Create(obj, info, properties)
    tween:Play()
    return tween
end

local function MakeDraggable(dragHandle, mainObject)
    local dragging = false
    local dragInput, dragStart, startPos

    dragHandle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = mainObject.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    dragHandle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            mainObject.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

function Neo:CreateGradient(parent, colors, rotation)
    local gradient = Instance.new("UIGradient")
    gradient.Rotation = rotation or 45
    local keypoints = {}
    for pos, color in pairs(colors) do
        table.insert(keypoints, ColorSequenceKeypoint.new(pos/100, color))
    end
    table.sort(keypoints, function(a, b) return a.Time < b.Time end)
    gradient.Color = ColorSequence.new(keypoints)
    gradient.Parent = parent
    return gradient
end

                local function UpdateValueFromInput(input)
                    local percent = math.clamp((input.Position.X - SlideBar.AbsolutePosition.X) / SlideBar.AbsoluteSize.X, 0, 1)
                    local rawVal = min + (max - min) * percent
                    value = tonumber(string.format("%."..rounding.."f", rawVal))
                    
                    ValLabel.Text = tostring(value)
                    CreateTween(Fill, TweenInfo.new(0.08), {Size = UDim2.new(percent, 0, 1, 0)})
                    CreateTween(Knob, TweenInfo.new(0.08), {Position = UDim2.new(percent, 0, 0.5, 0)})
                    
                    task.spawn(callback, value)
                end

                    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then

                    end
                end)

                UserInputService.InputChanged:Connect(function(input)
                    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then

                    end
                end)

                        dragging = false
                    end
                end)

                    SetValue = function(val)

                    end,
                    GetValue = function() return value end
                }
            end

                    SubBar.BackgroundColor3 = theme.Border
                    SubBar.ZIndex = 53
                    SubBar.Parent = ItemFrame

                    local SubFill = Instance.new("Frame")
                    SubFill.Size = UDim2.new((subVal - subMin) / (subMax - subMin), 0, 1, 0)
                    SubFill.BackgroundColor3 = theme.Accent
                    SubFill.BorderSizePixel = 0
                    SubFill.ZIndex = 54
                    SubFill.Parent = SubBar

                    local SubKnob = Instance.new("Frame")
                    SubKnob.Size = UDim2.new(0, 10, 0, 10)

                    SubKnob.BackgroundColor3 = theme.TextColor
                    SubKnob.ZIndex = 55
                    SubKnob.Parent = SubBar

                    local subDragging = false

                    local function UpdateSubVal(input)
                        local pct = math.clamp((input.Position.X - SubBar.AbsolutePosition.X) / SubBar.AbsoluteSize.X, 0, 1)
                        subVal = tonumber(string.format("%."..subRounding.."f", subMin + (subMax - subMin) * pct))
                        ItemVal.Text = tostring(subVal)
                        SubFill.Size = UDim2.new(pct, 0, 1, 0)
                        SubKnob.Position = UDim2.new(pct, 0, 0.5, 0)
                        task.spawn(subCallback, subVal)
                    end

                    SubBar.InputBegan:Connect(function(input)
                        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                            subDragging = true
                            UpdateSubVal(input)
                        end
                    end)

                    UserInputService.InputChanged:Connect(function(input)
                        if subDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                            UpdateSubVal(input)
                        end
                    end)

                    UserInputService.InputEnded:Connect(function(input)
                        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                            subDragging = false
                        end
                    end)

                    return {
                        SetValue = function(v)
                            subVal = math.clamp(v, subMin, subMax)
                            local pct = (subVal - subMin) / (subMax - subMin)
                            ItemVal.Text = tostring(subVal)
                            SubFill.Size = UDim2.new(pct, 0, 1, 0)
                            SubKnob.Position = UDim2.new(pct, 0, 0.5, 0)
                            task.spawn(subCallback, subVal)
                        end
                    }
                end

                return dropdownSecFunctions
            end

        if Themes[themeName] then
            Neo.CurrentTheme = Themes[themeName]
            theme = Neo.CurrentTheme

            CreateTween(MainStroke, TweenInfo.new(0.4), {Color = theme.Stroke})
            CreateTween(Topbar, TweenInfo.new(0.4), {BackgroundColor3 = theme.Sidebar})
            CreateTween(Sidebar, TweenInfo.new(0.4), {BackgroundColor3 = theme.Sidebar})
            CreateTween(SidebarLine, TweenInfo.new(0.4), {BackgroundColor3 = theme.Border})
            CreateTween(TopbarLine, TweenInfo.new(0.4), {BackgroundColor3 = theme.Border})
            CreateTween(Title, TweenInfo.new(0.4), {TextColor3 = theme.TextColor})
            CreateTween(SearchBoxFrame, TweenInfo.new(0.4), {BackgroundColor3 = theme.Card})
            CreateTween(SearchBoxStroke, TweenInfo.new(0.4), {Color = theme.Border})
            CreateTween(SearchInput, TweenInfo.new(0.4), {TextColor3 = theme.TextColor})
            CreateTween(UserProfile, TweenInfo.new(0.4), {BackgroundColor3 = theme.Card})
            CreateTween(Username, TweenInfo.new(0.4), {TextColor3 = theme.TextColor})
            CreateTween(UserRole, TweenInfo.new(0.4), {TextColor3 = theme.Accent})

            Logo.ImageColor3 = theme.Accent
        end
    end

    local isOpened = true
    
    local function ToggleUI()
        isOpened = not isOpened
        getgenv().UIToggled = isOpened
        if isOpened then
            MainFrame.Visible = true
            CreateTween(MainFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                Size = size,
                BackgroundTransparency = 0
            })
            CreateTween(FloatingBtn, TweenInfo.new(0.3), {BackgroundColor3 = theme.Sidebar})
        else
            local closeTween = CreateTween(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.In), {
                Size = UDim2.fromOffset(0, 0),
                BackgroundTransparency = 1
            })
            closeTween.Completed:Connect(function()
                if not isOpened then MainFrame.Visible = false end
            end)
            CreateTween(FloatingBtn, TweenInfo.new(0.3), {BackgroundColor3 = theme.Card})
        end
    end

    FloatingBtn.MouseButton1Click:Connect(ToggleUI)
    MinimizeBtn.MouseButton1Click:Connect(ToggleUI)

    CloseBtn.MouseButton1Click:Connect(function()
        Neo:ShowDialog({
            Title = "Exit Menu?",
            Content = "Are you sure you want to completely destroy and close this Neo interface?",
            Buttons = {
                {
                    Title = "Cancel",
                    Variant = "Secondary"
                },
                {
                    Title = "Exit",
                    Variant = "Primary",
                    Callback = function()
                        ScreenGui:Destroy()
                        OpenButtonGui:Destroy()
                        getgenv().NeoRunning = false
                        getgenv().AllControls = {}
                    end
                }
            }
        })
    end)

    SearchInput:GetPropertyChangedSignal("Text"):Connect(function()
        local query = string.lower(SearchInput.Text)
        for _, control in ipairs(getgenv().AllControls) do
            local ctrlName = string.lower(control.Name or "")
            local sectionNameLower = string.lower(control.SectionName or "")
            
            if query == "" then
                control.Element.Visible = true
                control.Section.Visible = true
            else
                if string.find(ctrlName, query) or string.find(sectionNameLower, query) then
                    control.Element.Visible = true
                    control.Section.Visible = true
                else
                    control.Element.Visible = false
                end
            end
        end
    end)
