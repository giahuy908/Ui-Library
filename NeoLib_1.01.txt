-- HÀM KHỞI TẠO ĐỐI TƯỢNG TỐI ƯU (CREATE HELPER)
local function Create(className, properties)
    local inst = Instance.new(className)
    for k, v in pairs(properties) do
        inst[k] = v
    end
    return inst
end

local function CreateTween(obj, info, properties)
    local tween = TweenService:Create(obj, info, properties)
    tween:Play()
    return tween
end

local function MakeDraggable(dragHandle, mainObject)
    local dragging, dragInput, dragStart, startPos = false, nil, nil, nil
    dragHandle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = mainObject.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)
    dragHandle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            mainObject.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)
end

function NeoLib:CreateGradient(parent, colors, rotation)
    local keypoints = {}
    for pos, color in pairs(colors) do
        table.insert(keypoints, ColorSequenceKeypoint.new(pos/100, color))
    end
    table.sort(keypoints, function(a, b) return a.Time < b.Time end)
    return Create("UIGradient", {
        Rotation = rotation or 45,
        Color = ColorSequence.new(keypoints),
        Parent = parent
    })
end

            function SectionControls:AddSlider(sliderText, min, max, default, rounding, callback)
                local value = math.clamp(default or min, min, max)
                rounding = rounding or 0

                local Slider = Create("Frame", {
                    Size = UDim2.new(1, 0, 0, 38),

                })
                local TitleLabel = Create("TextLabel", {

                    Size = UDim2.new(0, 50, 0, 16),

                    Font = Enum.Font.GothamBold,

                    TextColor3 = theme.Accent,
                    TextSize = 11,
                    TextXAlignment = Enum.TextXAlignment.Right,
                    Parent = Slider
                })

                })
                Create("UICorner", {CornerRadius = UDim.new(1, 0), Parent = SlideBar})

                local Fill = Create("Frame", {
                    Size = UDim2.new((value - min) / (max - min), 0, 1, 0),
                    BackgroundColor3 = theme.Accent,
                    BorderSizePixel = 0,
                    Parent = SlideBar
                })
                Create("UICorner", {CornerRadius = UDim.new(1, 0), Parent = Fill})

                local Knob = Create("Frame", {
                    Size = UDim2.new(0, 10, 0, 10),
                    AnchorPoint = Vector2.new(0.5, 0.5),
                    Position = UDim2.new((value - min) / (max - min), 0, 0.5, 0),
                    BackgroundColor3 = theme.TextColor,
                    Parent = SlideBar
                })
                Create("UICorner", {CornerRadius = UDim.new(1, 0), Parent = Knob})
                Create("UIStroke", {Color = theme.Accent, Thickness = 1.2, Parent = Knob})

                local dragging = false
                local function UpdateValueFromInput(input)
                    local percent = math.clamp((input.Position.X - SlideBar.AbsolutePosition.X) / SlideBar.AbsoluteSize.X, 0, 1)
                    value = tonumber(string.format("%."..rounding.."f", min + (max - min) * percent))

                    CreateTween(Fill, TweenInfo.new(0.08), {Size = UDim2.new(percent, 0, 1, 0)})
                    CreateTween(Knob, TweenInfo.new(0.08), {Position = UDim2.new(percent, 0, 0.5, 0)})

                end

                    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then

                    end
                end)
                UserInputService.InputChanged:Connect(function(input)
                    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                        UpdateValueFromInput(input)
                    end
                end)
                UserInputService.InputEnded:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then dragging = false end
                end)

                table.insert(getgenv().AllControls, {Name = sliderText, Element = Slider, Section = SectionFrame, SectionName = sectionName, TabName = tabName})
                return {
                    SetValue = function(val)
                        value = math.clamp(val, min, max)
                        local percent = (value - min) / (max - min)
                        ValLabel.Text = tostring(value)
                        CreateTween(Fill, TweenInfo.new(0.15), {Size = UDim2.new(percent, 0, 1, 0)})
                        CreateTween(Knob, TweenInfo.new(0.15), {Position = UDim2.new(percent, 0, 0.5, 0)})
                        task.spawn(callback, value)
                    end
                }
            end

    -- HỆ THỐNG HỘP THOẠI TRUY VẤN (DIALOG OVERLAY - GỌN GÀNG)
    function NeoLib:ShowDialog(config)
        local dialogTitle = config.Title or "Xác nhận"
        local dialogDesc = config.Content or ""
        local buttons = config.Buttons or {}

        local Overlay = Create("Frame", {
            Size = UDim2.new(1, 0, 1, 0),
            BackgroundColor3 = Color3.fromRGB(0, 0, 0),
            BackgroundTransparency = 1,
            ZIndex = 100,
            Parent = MainFrame
        })
        Create("TextButton", {Size = UDim2.new(1, 0, 1, 0), BackgroundTransparency = 1, Text = "", Parent = Overlay})

        local DialogFrame = Create("Frame", {
            Size = UDim2.fromOffset(260, 120),
            Position = UDim2.new(0.5, -130, 0.5, -60),
            BackgroundColor3 = theme.Card,
            Parent = Overlay
        })
        Create("UICorner", {CornerRadius = UDim.new(0, 6), Parent = DialogFrame})
        Create("UIStroke", {Color = theme.Border, Thickness = 1.2, Parent = DialogFrame})

        local Title = Create("TextLabel", {
            Size = UDim2.new(1, -16, 0, 26),
            Position = UDim2.new(0, 8, 0, 8),
            BackgroundTransparency = 1,
            Font = Enum.Font.GothamBold,
            Text = dialogTitle,
            TextColor3 = theme.TextColor,
            TextSize = 12,
            TextXAlignment = Enum.TextXAlignment.Left,
            Parent = DialogFrame
        })
        local Desc = Create("TextLabel", {
            Size = UDim2.new(1, -16, 0, 42),
            Position = UDim2.new(0, 8, 0, 34),
            BackgroundTransparency = 1,
            Font = Enum.Font.GothamMedium,
            Text = dialogDesc,
            TextColor3 = theme.SubText,
            TextSize = 11,
            TextWrapped = true,
            TextXAlignment = Enum.TextXAlignment.Left,
            Parent = DialogFrame
        })
        local BtnContainer = Create("Frame", {
            Size = UDim2.new(1, -16, 0, 24),
            Position = UDim2.new(0, 8, 1, -32),
            BackgroundTransparency = 1,
            Parent = DialogFrame
        })
        Create("UIListLayout", {
            FillDirection = Enum.FillDirection.Horizontal,
            HorizontalAlignment = Enum.HorizontalAlignment.Right,
            Padding = UDim.new(0, 6),
            Parent = BtnContainer
        })

        local function CloseDialog()
            CreateTween(Overlay, TweenInfo.new(0.2), {BackgroundTransparency = 1})
            local out = CreateTween(DialogFrame, TweenInfo.new(0.2, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Size = UDim2.fromOffset(0, 0), Position = UDim2.new(0.5, 0, 0.5, 0)})
            out.Completed:Connect(function() Overlay:Destroy() end)
        end

        for _, btnConfig in ipairs(buttons) do
            local Btn = Create("Frame", {
                Size = UDim2.new(0, 70, 1, 0),
                BackgroundColor3 = btnConfig.Variant == "Primary" and theme.Accent or theme.Background,
                Parent = BtnContainer
            })
            Create("UICorner", {CornerRadius = UDim.new(0, 5), Parent = Btn})

            local BtnLabel = Create("TextLabel", {
                Size = UDim2.new(1, 0, 1, 0),
                BackgroundTransparency = 1,
                Font = Enum.Font.GothamBold,
                Text = btnConfig.Title or "OK",
                TextColor3 = theme.TextColor,
                TextSize = 10,
                Parent = Btn
            })
            local BtnClick = Create("TextButton", {Size = UDim2.new(1, 0, 1, 0), BackgroundTransparency = 1, Text = "", Parent = Btn})

            BtnClick.MouseButton1Click:Connect(function()
                if btnConfig.Callback then task.spawn(btnConfig.Callback) end
                CloseDialog()
            end)
        end

        CreateTween(Overlay, TweenInfo.new(0.25), {BackgroundTransparency = 0.55})
        DialogFrame.Size = UDim2.fromOffset(0, 0)
        DialogFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
        CreateTween(DialogFrame, TweenInfo.new(0.25, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Size = UDim2.fromOffset(260, 120), Position = UDim2.new(0.5, -130, 0.5, -60)})
    end

    -- HOẠT ẢNH MỞ / ĐÓNG WINDOW CHÍNH
    local isOpened = true
    local function ToggleUI()
        isOpened = not isOpened
        if isOpened then
            MainFrame.Visible = true
            CreateTween(MainFrame, TweenInfo.new(0.35, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
                Size = size,
                BackgroundTransparency = 0
            })
            CreateTween(FloatingBtn, TweenInfo.new(0.25), {BackgroundColor3 = theme.Sidebar})
        else
            local closeTween = CreateTween(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Quart, Enum.EasingDirection.In), {
                Size = UDim2.fromOffset(0, 0),
                BackgroundTransparency = 1
            })
            closeTween.Completed:Connect(function()
                if not isOpened then MainFrame.Visible = false end
            end)
            CreateTween(FloatingBtn, TweenInfo.new(0.25), {BackgroundColor3 = theme.Card})
        end
    end

    -- TÌM KIẾM ĐỘNG TRÊN SIDEBAR
    SearchInput:GetPropertyChangedSignal("Text"):Connect(function()
        local query = string.lower(SearchInput.Text)
        for _, control in ipairs(getgenv().AllControls) do
            local ctrlName = string.lower(control.Name)
            local sectionNameLower = string.lower(control.SectionName)
            
            if query == "" then
                control.Element.Visible = true
                control.Section.Visible = true
            else
                if string.find(ctrlName, query) or string.find(sectionNameLower, query) then
                    control.Element.Visible = true
                    control.Section.Visible = true
                else
                    control.Element.Visible = false
                end
            end
        end
    end)

        Notify = function(_, config) return NeoLib:Notify(config) end
    }
