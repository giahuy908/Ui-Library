function NeoLib:Tween(object, time, properties, style)
	style = style or Enum.EasingStyle.Quad
	local tweenInfo = TweenInfo.new(time, style, Enum.EasingDirection.Out)
	local tween = TweenService:Create(object, tweenInfo, properties)
	tween:Play()
	return tween
end

function NeoLib:MakeDraggable(topbarobject, object)
	local Dragging = nil
	local DragInput = nil
	local DragStart = nil
	local StartPosition = nil

	local function UpdatePos(input)
		local Delta = input.Position - DragStart
		local pos = UDim2.new(StartPosition.X.Scale, StartPosition.X.Offset + Delta.X, StartPosition.Y.Scale, StartPosition.Y.Offset + Delta.Y)
		NeoLib:Tween(object, 0.15, { Position = pos }, Enum.EasingStyle.Cubic)
	end

	topbarobject.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			Dragging = true
			DragStart = input.Position
			StartPosition = object.Position

			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					Dragging = false
				end
			end)
		end
	end)

	topbarobject.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			DragInput = input
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if input == DragInput and Dragging then
			UpdatePos(input)
		end
	end)
end

				task.spawn(callback, activeValue)
			end

		end

					local realVal = math.round(percentage * 255)
					ValueNum.Text = tostring(realVal)
					NeoLib:Tween(Fill, 0.1, { Size = UDim2.new(percentage, 0, 1, 0) })
					onSlide(realVal)
				end

				Row.InputBegan:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
						dragging = true
						updateComp(input)
					end
				end)

				UserInputService.InputEnded:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
						dragging = false
					end
				end)

				UserInputService.InputChanged:Connect(function(input)
					if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
						updateComp(input)
					end
				end)
			end

			createRGBComponentSlider("R", math.round(defaultColor.R * 255), function(val)
				activeColor = Color3.fromRGB(val, math.round(activeColor.G * 255), math.round(activeColor.B * 255))
				triggerUpdate()
			end)

			createRGBComponentSlider("G", math.round(defaultColor.G * 255), function(val)
				activeColor = Color3.fromRGB(math.round(activeColor.R * 255), val, math.round(activeColor.B * 255))
				triggerUpdate()
			end)

			createRGBComponentSlider("B", math.round(defaultColor.B * 255), function(val)
				activeColor = Color3.fromRGB(math.round(activeColor.R * 255), math.round(activeColor.G * 255), val)
				triggerUpdate()
			end)

			PickerClick.MouseButton1Click:Connect(function()
				isOpened = not isOpened
				if isOpened then
					PickerPanel.Visible = true
					NeoLib:Tween(PickerPanel, 0.3, { Size = UDim2.new(1, 0, 0, 110) })
					NeoLib:Tween(cpStroke, 0.3, { Color = DefaultTheme.Accent })
				else
					NeoLib:Tween(PickerPanel, 0.3, { Size = UDim2.new(1, 0, 0, 0) })
					NeoLib:Tween(cpStroke, 0.3, { Color = DefaultTheme.Border })
					task.wait(0.3)
					PickerPanel.Visible = false
				end
			end)
		end

		return TabAPI
	end

	return WindowAPI
end

return NeoLib
